### 典型单元测试

```swift
func testCorrectHomophonesBasic() {
    let text = "这个配备有问题"
    let corrected = processor.correctHomophones(text: text)
    XCTAssertEqual(corrected, "这个配置有问题")
}

func testExportBatchEmptyResults() async {
    do {
        _ = try await service.exportBatch(results: [], ...)
        XCTFail("应该抛出 emptyResults 错误")
    } catch let error as BatchExportError {
        XCTAssertEqual(error, .emptyResults)
    }
}
```

### 典型集成测试

```swift
func testFullHistoryFlow() async throws {
    // 1. 创建转录结果
    let result = TranscriptionResult(...)
    
    // 2. 保存到历史记录
    let record = await repository.addHistoryRecord(from: result, ...)
    
    // 3. 验证保存
    let records = await repository.getAllHistory()
    XCTAssertTrue(records.contains(where: { $0.id == record.id }))
    
    // 4. 测试搜索
    let searchResults = await repository.searchHistory(query: "第一句")
    XCTAssertTrue(searchResults.contains(where: { $0.id == record.id }))
    
    // 5. 测试收藏
    await repository.toggleFavorite(id: record.id)
    let favorites = await repository.getFavorites()
    XCTAssertTrue(favorites.contains(where: { $0.id == record.id }))
    
    // 6. 测试删除
    await repository.deleteHistoryRecord(id: record.id, ...)
    let finalRecords = await repository.getAllHistory()
    XCTAssertFalse(finalRecords.contains(where: { $0.id == record.id }))
}
```
